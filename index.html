function renderTarjetas(tarjetasData, equipoPJTotal, goleadoresData) { // Creamos un mapa rápido para saber cuántos partidos lleva el jugador según la pestaña Goleadores const partidosJugadorMapa = {}; goleadoresData.forEach(g => { if(g.Nombre) partidosJugadorMapa[g.Nombre] = parseInt(g.Partidos) || 0; }); document.getElementById("body-tar").innerHTML = tarjetasData.filter(j => j.Nombre).map(j => { const amarillas = parseInt(j.Amarillas) || 0; const pjEquipo = equipoPJTotal[j.Equipo] || 0; const pjJugador = partidosJugadorMapa[j.Nombre] || 0; let estado = `✅ DISPONIBLE`; let pjAMostrar = pjEquipo; // LÓGICA DE AUTO-LIMPIEZA if (amarillas > 0 && amarillas % 3 === 0) { // Si el equipo y el jugador llevan los mismos partidos, // significa que el equipo aún no ha jugado el partido de castigo. if (pjEquipo <= pjJugador) { estado = `⚠️ SANCIONADO`; pjAMostrar = pjEquipo - 1; // Le restamos el que va a perder } else { // El equipo ya jugó (pjEquipo es mayor), así que ya cumplió. estado = `✅ DISPONIBLE (Sanción cumplida)`; pjAMostrar = pjJugador; // Se queda con los que jugó antes de la sanción } } return ` ${j.Nombre}
${j.Equipo} ${amarillas} ${pjAMostrar < 0 ? 0 : pjAMostrar} ${estado} `; }).join(''); }

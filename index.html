function renderPartidos(data) { const container = document.getElementById('partidos-container'); if (!data || data.length === 0) { container.innerHTML = "

No hay partidos registrados aún.

"; return; } container.innerHTML = data.map(m => `
🕒 ${m.Hora || '--:--'}
${m.Local || 'Equipo A'}
${m.GolesL || 0} - ${m.GolesV || 0}
${m.Visitante || 'Equipo B'}
${m.DetalleL && m.DetalleL !== '-' ? m.DetalleL : ''}
${m.DetalleV && m.DetalleV !== '-' ? m.DetalleV : ''}
`).join(''); }

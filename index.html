async function cargarDatos() { try { const [resP, resC, resG, resT] = await Promise.all([ fetch(`https://opensheet.elk.sh/${SHEET_ID}/Partidos`).then(r => r.json()), fetch(`https://opensheet.elk.sh/${SHEET_ID}/Clasificacion`).then(r => r.json()), fetch(`https://opensheet.elk.sh/${SHEET_ID}/Goleadores`).then(r => r.json()), fetch(`https://opensheet.elk.sh/${SHEET_ID}/Tarjetas`).then(r => r.json()) ]); if (resP) renderPartidos(resP); if (resC) renderClasificacion(resC); if (resG) renderGoleadores(resG); if (resT) renderTarjetas(resT); } catch (e) { console.error("Error:", e); } } function renderPartidos(data) { const container = document.getElementById('partidos-container'); if (!data || !Array.isArray(data)) return; container.innerHTML = data.map(m => `
🕒 HORA: ${m.Hora || '--:--'}
${m.Local || '---'}
${m.GolesL || 0} - ${m.GolesV || 0}
${m.Visitante || '---'}
${m.DetalleL || ''}
${m.DetalleV || ''}
`).join(''); } function renderTarjetas(data) { const tbody = document.querySelector("#tabla-tar tbody"); if (!tbody) return; tbody.innerHTML = data.map(j => { // Usamos 'Amarillas' y 'Partidos' tal cual están en tu Excel const am = parseInt(j.Amarillas) || 0; const pj = j.Partidos || 0; const sancionado = am > 0 && am % 3 === 0; return ` ${j.Nombre || '---'}
${j.Equipo || ''} ${am} ${pj} ${sancionado ? '⚠️ SANCIONADO' : '✅ DISPONIBLE'} `; }).join(''); }
